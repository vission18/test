//消费类别表
CREATE TABLE MMS_XF_TYPE ( 
    TYPE_ID   	VARCHAR(100) NOT NULL,
    TYPE_NAME  	VARCHAR(100),
    TYPE_STATUS	CHARACTER(1),
    REMARK    	VARCHAR(512) 
    )
GO

//消费记录表
CREATE TABLE MMS_XF_RECORD ( 
    XF_ID    	VARCHAR(100) NOT NULL primary key,
    XF_TYPE  	VARCHAR(100),
    XF_GOODS	VARCHAR(100),
    XF_PRICE	INT,
    XF_DATE	 VARCHAR(10),
    RECORD_DATE	VARCHAR(20),
    RECORD_USER	VARCHAR(100),
    XF_DETAIL     	VARCHAR(512) 
    )
GO
